CREATE TABLE "SYSTEM"."USER"
(    "UNAME" VARCHAR2(20 BYTE) NOT NULL ENABLE,
      "PWD" VARCHAR2(70 BYTE),
      "STATUS" NUMBER(1,0),
      CONSTRAINT "USER_PK" PRIMARY KEY("UNAME"));

CREATE TABLE "SYSTEM"."USERS_ROLES"
 (   "ROLE" VARCHAR2(20 BYTE),
     "UNAME" VARCHAR2(20 BYTE) ,
      CONSTRAINT "FK1" FOREIGN KEY("UNAME")
      REFERENCES "USER"("UNAME") ENABLE);
